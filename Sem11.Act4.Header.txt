void generar_valores(int* arreglo, int& N)
{
	N = rand() % 51 + 1;

	arreglo = new int[N];

	for (int i = 0; i < N; i++)
	{
		arreglo[i] = rand() % 12 + 40;
	}

	for (int i = 0; i < N; i++)
	{
		cout << "Arreglo[" << i << "]: " << arreglo[i] << endl;
	}

	system("pause>0");

	int count = 0;

	for (int i = 0; i < N; i++)
	{
		if (arreglo[i] % 3 != 0)
		{
			arreglo[count++] = arreglo[i];
			//para que cuente que arreglos no son multi de 3
		}

	}
	N = count;
	cout << "Arreglo sin multiplo de 3: " << endl;
	for (int i = 0; i < N; i++)
	{
		cout << "Arreglo[" << i << "]: " << arreglo[i] << endl;
	}

	system("pause>0");

	int aux;
	for (int i = 0; i < N; i++)
	{
		for (int j = i + 1; j < N; j++)
		{
			if (arreglo[i] < arreglo[j])
			{
				aux = arreglo[i];
				arreglo[i] = arreglo[j];
				arreglo[j] = aux;
			}
		}
	}
	//Imprimir valores

	cout << "Mayor a menor: " << endl;
	for (int i = 0; i < N; i++)
	{
		cout << "Arreglo[" << i << "]: " << arreglo[i] << endl;
	}
	system("pause>0");

}

void agregar_numero(int*& arreglo, int& N, int num)
{
	int* nuevo_arreglo = new int[N + 1];
	int i = 0;
	while (i < N && arreglo[i] < num)
	{
		nuevo_arreglo[i] = arreglo[i];
		i++;
	}
	nuevo_arreglo[i] = num;
	while (i < N)
	{
		nuevo_arreglo[i + 1] = arreglo[i];
		i++;
	}
	N++;
	delete[] arreglo;
	arreglo = nuevo_arreglo;
	cout << "Arreglo después de agregar " << num << ":" << endl;
	for (int i = 0; i < N; i++) {
		cout << "Arreglo[" << i << "]: " << arreglo[i] << endl;
	}
}
