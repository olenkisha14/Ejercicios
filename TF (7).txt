void dibujar_fondo()
{
    for (int f = 0; f < 69; f++) {
        for (int c = 0; c < 170; c++) {
            if (fondo[f][c] == 0) Console::ForegroundColor = ConsoleColor::Black;
            if (fondo[f][c] == 1) Console::ForegroundColor = ConsoleColor::White;
            if (fondo[f][c] == 2) FontColor(255, 192, 0);
            cout << (char)219;
        }
        cout << endl;
    }
}

void funcionalidad_naves()
{
    while (true)
    {
        system("cls");
        Console::BackgroundColor = ConsoleColor::Black;

        int salud_enemigo = 3;
        int salud_jugador = 3;
        int x = 50, y = 59;
        int x1 = 15, y1 = 5, dx1 = 1;
        dibujar_fondo();
        dibujar_nave(x, y);
        dibujar_naveenemiga(x1, y1);
        dibujar_panel_de_control(170, 0);
        dibujar_saludenemigo(salud_enemigo);
        dibujar_saludjugador(salud_jugador);
        Console::BackgroundColor = ConsoleColor::White;
        Console::ForegroundColor = ConsoleColor::Black;
        cursor(178, 14); cout << "*SALUD NAVE ENEMIGA*" << endl;
        cursor(182, 28); cout << "*SALUD NAVE*" << endl;

        BalaJugador* balas_jugador[100]; // arreglo para los proyectiles del jugador
        int num_balas = 0;

        BalaEnemiga* balas_enemigas[100]; // arreglo para las balas de las naves enemigas
        int num_balas_enemigas = 0;

        bool juego_terminado = false;

        while (!juego_terminado)
        {
      
            if (_kbhit())
            {

                //borrar nave
                borrar_nave(x, y);
                //mover nave
                char tecla = getch();
                if (tecla == 'd') x++;
                if (tecla == 'a') x--;
                if (tecla == 'w') y--;
                if (tecla == 's') y++;
                if (tecla == 32)
                {
                    balas_jugador[num_balas] = new BalaJugador(x + 5, y - 1, -1);
                    num_balas++;

                    balas_jugador[num_balas] = new BalaJugador(x + 9, y - 1, -1);
                    num_balas++;
                }
                //dibujar nave
                dibujar_nave(x, y);

            }
            for (int i = 0; i < num_balas; i++) {
                //ANIMACION DE LA BALA DEL JUGADOR
                //borrar
                cursor(balas_jugador[i]->x, balas_jugador[i]->y); cout << " ";
                //mover
                if (balas_jugador[i]->y < 1)
                {
                    balas_jugador[i]->dy = 0;
                    balas_jugador[i]->y = 0; // Aseguramos que el proyectil no se dibuje fuera de la consola
                }
                else {
                    balas_jugador[i]->y = balas_jugador[i]->y + balas_jugador[i]->dy;
                }
                //dibujar
                if (balas_jugador[i]->y > 0) { // Solo dibujamos el proyectil si está dentro de la consola
                    balas_jugador[i]->dibujar_bala();
                }
                // Verificar colisión con la nave enemiga
                if (colision(balas_jugador[i]->x, balas_jugador[i]->y, x1, y1, 10, 11))
                {
                    disminuir_salud_enemigo(&salud_enemigo);
                    dibujar_saludenemigo(salud_enemigo);
                    delete balas_jugador[i];
                    for (int j = i; j < num_balas - 1; j++) {
                        balas_jugador[j] = balas_jugador[j + 1];
                    }
                    num_balas--;
                    i--;

                    // Verificar si la salud del enemigo llegó a 0
                    if (salud_enemigo == 0) {
                        cursor(80, 35); cout << "YOU WIN!";
                        _sleep(2000); // Pausa para mostrar el mensaje
                        juego_terminado = true;
                        return; // Salir del juego
                    }
                }

            }

            if (juego_terminado) break;

            //borrar nave enemiga
            borrar_naveenemiga(x1, y1);
            //movimiento nave enemiga
            if (x1 > 130 || x1 < 10) {
                dx1 = dx1 * -1;
            }
            x1 += dx1;
            //dibujar nave enemiga
            dibujar_naveenemiga(x1, y1);
            // Añadir balas enemigas de forma aleatoria o controlada por tiempo
            if (rand() % 10 == 0) {
                if (num_balas_enemigas < 100) {
                    balas_enemigas[num_balas_enemigas] = new BalaEnemiga(x1 + 12, y1 + 8, 1);
                    num_balas_enemigas++;
                }
                if (num_balas_enemigas < 100) {
                    balas_enemigas[num_balas_enemigas] = new BalaEnemiga(x1 + 16, y1 + 8, 1);
                    num_balas_enemigas++;
                }
            }
            for (int i = 0; i < num_balas_enemigas; i++) {
                //borrar
                cursor(balas_enemigas[i]->x, balas_enemigas[i]->y); cout << " ";
                //mover
                balas_enemigas[i]->y += balas_enemigas[i]->dy;
                // Verificar si la bala está fuera de los límites
                if (balas_enemigas[i]->y > 69) {
                    // Eliminar ddla bala del arreglo y liberar memoria
                    delete balas_enemigas[i];
                    for (int j = i; j < num_balas_enemigas - 1; j++) {
                        balas_enemigas[j] = balas_enemigas[j + 1];
                    }
                    num_balas_enemigas--;
                    i--;
                }
                else {
                    // Dibujar si está dentro de la consola
                    balas_enemigas[i]->dibujar_balaenemiga();
                }
                // Verificar colisión con la nave del jugador
                if (colision(balas_enemigas[i]->x, balas_enemigas[i]->y, x, y, 10, 5))
                {
                    disminuir_salud_jugador(&salud_jugador);
                    dibujar_saludjugador(salud_jugador);
                    delete balas_enemigas[i];
                    for (int j = i; j < num_balas_enemigas - 1; j++) {
                        balas_enemigas[j] = balas_enemigas[j + 1];
                    }
                    num_balas_enemigas--;
                    i--;

                    }
                }
            }
